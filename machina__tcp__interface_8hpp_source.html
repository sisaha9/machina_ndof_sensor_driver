47  /**
48  * @brief Start a TCP connection with the sensor
49  *
50  * @param sensor_ip IP address of the sensor
51  * @param port Port number of the sensor
52  */
53  bool StartConnection(const std::string& sensor_ip, uint16_t port);
54  /**
55  * @brief Send data to the sensor
56  *
57  * Doesn't do any validation and assumes the data is valid and len is correct
58  * @param data Pointer to the data to be sent
59  * @param len Length of the data to be sent
60  * @return true If the data was sent successfully
61  */
62  bool SendData(uint8_t* data, uint64_t len);
63  /**
64  * @brief Send data to the sensor with a timeout
65  *
66  * Same assumptions as SendData
67  *
68  * @param data Pointer to the data to be sent
69  * @param len Length of the data to be sent
70  * @param timeout Timeout in microseconds
71  * @return true If the data was sent successfully
72  */
73  bool SendData(uint8_t* data, uint64_t len, uint64_t timeout);
74  /**
75  * @brief Receive data from the sensor
76  *
77  * Assumes the buffer is valid and len is correct
78  *
79  * @param data Pointer to the buffer to store the received data
80  * @param len Length of the data to be received
81  * @return uint64_t Number of bytes received
82  */
83  uint64_t ReceiveData(uint8_t* data, uint64_t len);
84  /**
85  * @brief Receive data from the sensor with a timeout

88  *
89  * @param data Pointer to the buffer to store the received data
90  * @param len Length of the data to be received
91  * @param timeout Timeout in microseconds

94  uint64_t ReceiveData(uint8_t* data, uint64_t len, uint64_t timeout);

100  bool IsSocketConnected();

107
